Descifrador MD5, Base64 y SHA1

function descifrar() { const texto = document.getElementById('input').value; const algoritmo = document.getElementById('algorithm').value; if (!texto || !algoritmo) { alert('Por favor, ingrese el texto y seleccione un algoritmo.'); return; } let resultado; switch (algoritmo) { case 'md5': resultado = CryptoJS.MD5(texto).toString(); break; case 'base64': resultado = atob(texto); break; case 'sha1': resultado = CryptoJS.SHA1(texto).toString(); break; default: alert('Algoritmo no v√°lido.'); return; } document.getElementById('result').textContent = `Resultado: ${resultado}`; }
